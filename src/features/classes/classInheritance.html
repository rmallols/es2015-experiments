//Classes can extend other classes, as we used to do with Object.create
    class Rectangle extends Shape {
        constructor (id, x, y, width, height) {
            super(id, x, y) //calls the constructor of the 'super' == parent classes
            //This is overriding the width & height props following classical prototypical inheritance
            this.width  = width
            this.height = height
        }
    }
    class Circle extends Shape {
        constructor (id, x, y, radius) {
            super(id, x, y)
            //This is adding a new property to the Circle class, not available on the parent (Shape) class
            this.radius = radius
        }
    }

    var Rectangle = function (id, x, y, width, height) {
        Shape.call(this, id, x, y);
        this.width  = width;
        this.height = height;
    };
    Rectangle.prototype = Object.create(Shape.prototype);
    Rectangle.prototype.constructor = Rectangle;
    var Circle = function (id, x, y, radius) {
        Shape.call(this, id, x, y);
        this.radius = radius;
    };
    Circle.prototype = Object.create(Shape.prototype);
    Circle.prototype.constructor = Circle;
